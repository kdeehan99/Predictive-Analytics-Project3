from sklearn.metrics import roc_curve, auc
# Step 1: Convert predictions to lists
y_true = []
y_scores = []

# Define threshold for relevance (you can adjust this)
relevance_threshold = 4.0

for pred in preds_ubcf:
    true_rating = pred.r_ui
    predicted_rating = pred.est

    # Binary relevance: 1 if true rating >= threshold, else 0
    y_true.append(1 if true_rating >= relevance_threshold else 0)
    y_scores.append(predicted_rating)  # Use predicted score

# Step 2: ROC Curve
fpr, tpr, _ = roc_curve(y_true, y_scores)

def get_top_n(predictions, n=10, min_rating=4):
    '''Return {uid: [book_id,]}  of the N highest predicted ratings  min_rating.'''
    from collections import defaultdict
    top_n = defaultdict(list)
    for uid, iid, true_r, est, _ in predictions:
        if est >= min_rating:
            top_n[uid].append( (iid, est) )
    for uid, user_ratings in top_n.items():
        user_ratings.sort(key=lambda x: x[1], reverse=True)
        top_n[uid] = [iid for (iid, _) in user_ratings[:n]]
    return top_n
